	int SumAfterN(int n) {

是否是素数：

	int isPrime(int n){
	int i,sign=1;
	if(n<2)
		sign=0;
	for(i=2;i<=sqrt(n);i++){
	if(n%i==0){
		sign=0;
	}
	}
	
	return sign;
	}
